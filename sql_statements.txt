CREATE TABLE `estate_users`.`personnel_list` ( `table_id` INT(4) NOT NULL AUTO_INCREMENT , `firstname` VARCHAR(50) NOT NULL , `lastname` VARCHAR(50) NOT NULL , `email` VARCHAR(30) NOT NULL , `phone` VARCHAR(15) NOT NULL , `home_address` VARCHAR(40) NOT NULL , `nationality` VARCHAR(20) NOT NULL , `qualification` TEXT NOT NULL , `spoken_language` TEXT NOT NULL , `employment_date` VARCHAR(10) NOT NULL , `branch` VARCHAR(20) NOT NULL , `department` VARCHAR(20) NOT NULL , `position` VARCHAR(30) NOT NULL , `contact_person` VARCHAR(40) NOT NULL , `contact_address` VARCHAR(30) NOT NULL , `contact_phone` VARCHAR(20) NOT NULL , `contact_relation` VARCHAR(20) NOT NULL , `note` TEXT NOT NULL , `gender` VARCHAR(8) NOT NULL , `profile_picture` TEXT NOT NULL , PRIMARY KEY (`table_id`)) ENGINE = InnoDB;
